					$.each(data, function(index, value) {
						if(value.artist && value.title) {
							$('#results').append('<li class="result" id="' + value.postid + '">' + value.artist + " - " + value.title + '</li>');
							$.getJSON("https://gdata.youtube.com/feeds/api/videos?q=" + value.artist.replace(" ", "+") + "+" + value.title.replace(" ", "+") + "&max-results=1&v=2&alt=json", function(youtube_res) {
								console.log(youtube_res);
								data = youtube_res.feed;
								if(!data.entry) {
									return;
								}
								videoid = data.entry[0].media$group.yt$videoid.$t;
								console.log(videoid);
								$('#' + value.postid).append('<iframe height="30" src="http://www.youtube.com/embed/' + videoid + '" frameborder="0" allowfullscreen></iframe>');								
							});
						}
					})
				});
